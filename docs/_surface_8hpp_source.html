218

220  const PosIdx pos_idx_max = PosIdx(pos_window_size.prod());

468  VecDf ray(const VecDf& pos_origin_, const VecDf& dir_) const
469  {
470 
471  using ChildHits = std::vector<ChildHit>;
472 
473  // If ray is cast from within isogrid grid, first check child grid containing origin point.
474  if (m_grid_isogrid.inside(pos_origin_))
475  {
476  const VecDf& pos_hit = ray(
477  pos_origin_, dir_,
478  m_grid_isogrid.children().get(
479  m_grid_isogrid.pos_child(pos_origin_.template cast<NodeIdx>())
480  )
481  );
482  if (pos_hit != ray_miss)
483  return pos_hit;
484  }
485 
486  // Ray to test against.
487  Line line(pos_origin_, dir_);
488 
489  // Tracking list for child grids that are hit.
490  ChildHits child_hits;
491 
492  // Cycle each axis, casting ray to child grid planes marching away from origin.
493  for (Dim dim = 0; dim < dir_.size(); dim++)
494  {
495  // Direction +/-1 along this axis.
496  Distance dir_dim = Distance(sgn(dir_(dim)));
497  if (dir_dim == 0)
498  continue;
499 
500  // Get next child plane along this axis.
501  Distance pos_plane_dim = round_to_next(
502  dim, dir_dim, pos_origin_(dim), m_grid_isogrid.child_size()
503  );
504 
505  // Construct vector with elements not on this axis at zero.
506  VecDf pos_plane = VecDf::Constant(0);
507  pos_plane(dim) = pos_plane_dim;
508 
509  // If the zero point on this plane is not within the grid, then jump to max/min point
510  // on isogrid grid.
511  if (!m_grid_isogrid.inside(pos_plane))
512  {
513  Distance pos_grid_dim;
514  // If casting in -'ve direction, get maximum extent.
515  if (dir_dim == -1)
516  {
517  pos_grid_dim = Distance(
518  m_grid_isogrid.offset()(dim) + m_grid_isogrid.size()(dim)
519  );
520  if (pos_plane_dim < pos_grid_dim)
521  continue;
522  }
523  // Else if casting in +'ve direction, get minimum extent.
524  else
525  {
526  pos_grid_dim = Distance(m_grid_isogrid.offset()(dim));
527  if (pos_plane_dim > pos_grid_dim)
528  continue;
529  }
530  // Reset plane position to max/min extent as calculated above.
531  pos_plane(dim) = pos_grid_dim;
532  }
533 
534  // Plane normal is opposite to ray direction.
535  VecDf normal = VecDf::Constant(0);
536  normal(dim) = -dir_dim;
537 
538  // Cast ray to plane and track any child grids hit on the way to tracking list.
539  // If child size is not a factor of grid size then this first cast could be to outside
540  // the grid. So cannot quit early here and must try next child.
541  ray_check_track_child(child_hits, line, Plane(normal, pos_plane(dim) * dir_dim));
542 
543  // Round up/down to next child, in case we started at inexact modulo of child grid size
544  // (i.e. when isogrid grid size is not integer multiple of child grid size).
545  pos_plane_dim = round_to_next(
546  dim, dir_dim, pos_plane(dim), m_grid_isogrid.child_size()
547  );
548  // If rounding produced a different plane, then cast to that plane, and potentially track
549  // child grid to tracking list.
550  if (pos_plane_dim != pos_plane(dim))
551  {
552  pos_plane(dim) = pos_plane_dim;
553  if (!ray_check_track_child(child_hits, line, Plane(normal, pos_plane(dim) * dir_dim)))
554  continue;
555  }
556 
557  // Keep marching along planes, casting ray to each and tracking any candidate child
558  // grids to the tracking list.
559  const Distance child_size_dim = Distance(m_grid_isogrid.child_size()(dim));
560  while (true)
561  {
562  pos_plane(dim) += dir_dim * child_size_dim;
563  if (
564  !ray_check_track_child(
565  child_hits, line, Plane(normal, pos_plane(dim) * dir_dim)
566  )
567  ) {
568  break;
569  }
570  }
571  }
572 
573  // untrack children whose raycast hit point lies outside themselves. Possible if a child
574  // has been allowed to be tracked in order for the loop above to continue on to the next
575  // child, via dir vs. offset/size check in ray_check_track_child().
576 // child_hits.erase(std::untrack_if(
577 // child_hits.begin(), child_hits.end(),
578 // [this](const ChildHit& child_hit) {
579 // return !this->m_grid_isogrid.children()
580 // .get(child_hit.pos_child)
581 // .inside(child_hit.pos_intersect);
582 // }
583 // ), child_hits.end());
584 
585  // Sort candidate child grids in distance order from front to back.
586  std::sort(
587  child_hits.begin(), child_hits.end(),
588  [&pos_origin_](const ChildHit& a, const ChildHit& b) -> bool {
589  const VecDf& dist_a = (a.pos_intersect - pos_origin_);
590  const VecDf& dist_b = (b.pos_intersect - pos_origin_);
591  return dist_a.squaredNorm() < dist_b.squaredNorm();
592  }
593  );
594  // untrack any duplicate child grids from the sorted list (i.e. where ray intersects
595  // precisely at the interesction of two or more planes).
596  child_hits.erase(std::unique(
597  child_hits.begin(), child_hits.end(),
598  [](const ChildHit& a, const ChildHit& b) -> bool {
599  return a.pos_idx_child == b.pos_idx_child;
600  }
601  ), child_hits.end());
602 
603  // For each candidate child, cast ray through until the zero-curve is hit.
604  for (const ChildHit& child_hit : child_hits)
605  {
606  const VecDf& pos_hit = ray(
607  child_hit.pos_intersect, dir_,
608  m_grid_isogrid.children().get(child_hit.pos_idx_child)
609  );
610 
611  if (pos_hit != ray_miss)
612  return pos_hit;
613  }
614 
615  return ray_miss;
616  }

820  // Will initialise to zero, so no further work needed for
821  // these new indices giving the start of the range.
822  aidx_first_neigh[ListIdx(layer_idx)].resize(num_childs);
823  // The final index needs to be copied from the current size
824  // of each spatial partition, so loop over partitions,
825  // copying their size into the respective last index list.
826  for (
827  ListIdx list_idx_child = 0; list_idx_child < num_childs; list_idx_child++
828  ) {
829  // Get position of this spatial partition in parent
830  // lookup grid.
831  const PosIdx pos_idx_child =
832  m_grid_affected.children().lookup().list(layer_idx)[list_idx_child];
833  // Copy number of active grid nodes for this partition
834  // into relevant index in the list.
835  aidx_last_neigh[ListIdx(layer_idx)][list_idx_child] =
836  m_grid_affected.children().get(pos_idx_child).list(layer_idx).size();
837  }
838  }
839 
840  // Loop each layer finding the affected outer layer points
841  // for each partition using the start and end points cached
842  // above.
843 
844  for (LayerId layer_id = s_layer_min; layer_id <= s_layer_max; layer_id++)
845  {
846  const TupleIdx layer_idx = this->layer_idx(layer_id);
847 
848  // Loop over spatial partitions, ignoring newly tracked ones
849  // since we're using the cached spatial partition list as
850  // the end of the range.
851  for (
852  ListIdx idx_child = 0;
853  idx_child < aidx_first_neigh[ListIdx(layer_idx)].size();
854  idx_child++
855  ) {
856  // Get position of this spatial partition in this layer.
857  const PosIdx pos_idx_child = (
858  m_grid_affected.children().lookup().list(layer_idx)[idx_child]
859  );
860 
861  // Loop over leaf grid nodes within this spatial partition, using the cached
862  // start and end indices, so that newly tracked points are skipped.
863  for (
864  ListIdx idx_neigh = aidx_first_neigh[ListIdx(layer_idx)][idx_child];
865  idx_neigh < aidx_last_neigh[ListIdx(layer_idx)][idx_child];
866  idx_neigh++
867  ) {
868  using AffectedChild = typename AffectedLookupGrid::Child;
869  const AffectedChild& child = m_grid_affected.children().get(pos_idx_child);
870  // This leaf grid node is the centre to search about.
871  const PosIdx pos_idx_centre = child.list(layer_idx)[idx_neigh];
872  const VecDi pos_centre = child.index(pos_idx_centre);
873 
874  // Cycle neighbours and record them if they are within the narrow band.
875  m_grid_isogrid.neighs(
876  pos_centre,
877  [this](const VecDi& pos_neigh) {
878  // Calculate layer of this neighbouring point from the isogrid grid.
879  const LayerId layer_id_neigh = this->layer_id(pos_neigh);
880  // If the calculated layer lies within the narrow band, then we
881  // want to track it.
882  if (inside_band(layer_id_neigh))
883  {
884  // track the neighbour point to the
885  // tracking grid. Will reject
886  // duplicates.
887  m_grid_affected.track(
888  pos_neigh, this->layer_idx(layer_id_neigh)
889  );
890  }
891  }
892  ); // End neighbourhood query.
893  } // End for leaf grid node.
894  } // End for spatial partition.
895  } // End for layer.
896 
897  // Now we've found the neighbours of the pre-existing grid
898  // points in the tracking list, we want to skip these on the
899  // next loop, so set the start index for each partition of
900  // each layer to the previous end index.
901 
902  for (LayerId layer_id = s_layer_min; layer_id <= s_layer_max; layer_id++)
903  {
904  const ListIdx layer_idx = ListIdx(this->layer_idx(layer_id));
905  // Loop over spatial partitions.
906  for (ListIdx idx = 0; idx < aidx_first_neigh[layer_idx].size(); idx++)
907  // Set first index in spatial partition's tracking list
908  // to be the previous last index, so we start there on
909  // next loop around.
910  aidx_first_neigh[layer_idx][idx] = aidx_last_neigh[layer_idx][idx];
911  }
912  }
913  } // End calc_affected.

1387  Distance distance (VecDi pos_, Distance dist_, const Distance dir_) const
1388  {
1389  #ifdef FELT_DEBUG_ENABLED
1390  const VecDi pos_original(pos_);
1391  #endif
1392  // Transform to unsigned distance.
1393  dist_ *= dir_;
1394 
1395  // Get neighbouring point that is next closest to the zero-layer.
1396  m_grid_isogrid.neighs(
1397  std::move(pos_),
1398  [this, &dist_, dir_](const VecDi& pos_neigh_) {
1399  if (not m_grid_isogrid.inside(pos_neigh_))
1400  return;
1401  const Distance dist_neigh = m_grid_isogrid.get(pos_neigh_);
1402  // Check absolute value of this neighbour is less than nearest point.
1403  // Multiplying by `dir` has two effects:
1404  // - It has the same effect as abs() for points on the same side of the band.
1405  // - It ensures points on the opposite side are negative, so that < comparison
1406  // prefers those points, which is good because we're interested in the neighbour
1407  // in the *direction* of the zero-curve.
1408  if (dist_neigh*dir_ < dist_) {
1409  dist_ = dist_neigh*dir_;
1410  }
1411  }
1412  );
1413 
1414  // This point's distance is then the distance of the closest neighbour +/-1, depending
1415  // which side of the band we are looking at. So first transform back into signed distance
1416  // then add +/-1.
1417  const Distance dist_neigh = dist_*dir_;
1418  dist_ = dist_neigh + dir_;
1419 
1420  #ifdef FELT_DEBUG_ENABLED
1421  const LayerId layer_id_pos = layer_id(pos_original);
1422  const LayerId layer_id_neigh = layer_id(dist_neigh);
1423 
1424  if (
1425  std::abs(layer_id_pos) < std::abs(layer_id_neigh) &&
1426  sgn(layer_id_pos) == sgn(layer_id_neigh)
1427  ) {
1428  std::stringstream sstr;
1429  sstr << "Neighbour closest to zero curve is further away than source position: " <<
1430  Felt::format(pos_original) << " at " << m_grid_isogrid.get(pos_original) <<
1431  " is closer than " << Felt::format(pos_) << " at " << m_grid_isogrid.get(pos_) <<
1432  " but should not be";
1433  std::string str = sstr.str();
1434  throw std::domain_error(str);
1435  }
1436 
1437  #endif
1438 
1439  return dist_;
