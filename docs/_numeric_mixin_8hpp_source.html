403  Leaf interp (const VecDf& pos_) const
404  {
405  const VecDi& size = pself->size();
406  const VecDi& pos_floor = pos_.array().floor().matrix().template cast<NodeIdx>();
407 
408  // Store all 2^d corners.
409  DataArray< Leaf > val_corners(PosIdx(1 << size.size()));
410 
411  // Get all corners of containing cell.
412  for (ListIdx i = 0; i < val_corners.size(); i++)
413  {
414  // 0 = 00 => (x,y)
415  // 1 = 01 => (x+1,y)
416  // 2 = 10 => (x,y+1)
417  // 3 = 11 => (x+1,y+1)
418 
419  VecDi pos_corner(pos_floor);
420  for (Dim axis = 0; axis < pos_corner.size(); axis++)
421  {
422  const NodeIdx dir = (i >> axis) & 1;
423  pos_corner(axis) += dir;
424  }
425 
426  val_corners[i] = pself->get(pos_corner);
427  }
428 
429  // Translate position vector into 'hypercube space',
430  // so 0 <= v(x) <= 1.
431  VecDf pos_centred = pos_ - pos_.array().floor().matrix();
432 
433  // Repeatedly reduce along axes,
434  // i.e. hypercube -> cube -> square -> line -> point
435  while (val_corners.size() > 1)
436  interp(val_corners, pos_centred);

452  {
453  const ListIdx num_corners = val_corners_.size();
454 
455  // Number of values returned.
456  // This is a power of 2 less than input dimensions
457  // (cube becomes square, square becomes line, line becomes point).
458  const ListIdx num_out = num_corners >> 1;
459 
460  static_assert(

462  "num_corners must be unsigned to use __builtin_ctzl" //__builtin_ffsl
463  );

465  // This is computed from the dimensions of the original input and
466  // the dimensions of the intended output.
467  const Eigen::Index axis_idx = pos_.size() - Eigen::Index(__builtin_ctzl(num_corners));
468 
469  // The weighting to be used in interpolating each pair of points.
470  // This is the position along the axis of interpolation.

472
