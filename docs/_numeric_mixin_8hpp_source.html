397  LeafType interp (const VecDf& pos_) const
398  {
399  const VecDi& size = pself->size();
400  const VecDi& pos_floor = pos_.array().floor().matrix().template cast<INT>();
401 
402  // Store all 2^d corners.
403  DataArray< LeafType > val_corners(PosIdx(1 << size.size()));
404 
405  // Get all corners of containing cell.
406  for (ListIdx i = 0; i < val_corners.size(); i++)
407  {
408  // 0 = 00 => (x,y)
409  // 1 = 01 => (x+1,y)
410  // 2 = 10 => (x,y+1)
411  // 3 = 11 => (x+1,y+1)
412 
413  VecDi pos_corner(pos_floor);
414  for (Dim axis = 0; axis < pos_corner.size(); axis++)
415  {
416  const INT dir = (i >> axis) & 1;
417  pos_corner(axis) += dir;
418  }
419 
420  val_corners[i] = pself->get(pos_corner);
421  }
422 
423  // Translate position vector into 'hypercube space',
424  // so 0 <= v(x) <= 1.
425  VecDf pos_centred = pos_ - pos_.array().floor().matrix();
426 
427  // Repeatedly reduce along axes,
428  // i.e. hypercube -> cube -> square -> line -> point
429  while (val_corners.size() > 1)
430  interp(val_corners, pos_centred);
431 
432  return val_corners[0];
433  }

445  static void interp (DataArray<LeafType>& val_corners_, const VecDf& pos_)
446  {
447  const ListIdx num_corners = val_corners_.size();
448 
449  // Number of values returned.
450  // This is a power of 2 less than input dimensions
451  // (cube becomes square, square becomes line, line becomes point).
452  const ListIdx num_out = num_corners >> 1;
453 
454  static_assert(
455  std::is_same<ListIdx, unsigned long>::value,
456  "num_corners must be unsigned to use __builtin_ctzl" //__builtin_ffsl
457  );
458  // The axis along which to interpolate.
459  // This is computed from the dimensions of the original input and
460  // the dimensions of the intended output.
461  const Dim axis_idx = pos_.size() - Dim(__builtin_ctzl(num_corners));
462 
463  // The weighting to be used in interpolating each pair of points.
464  // This is the position along the axis of interpolation.
465  const FLOAT axis_pos = pos_(axis_idx);
466 
467  for (ListIdx i = 0; i < num_out; i++)
468  {
469  const LeafType low = val_corners_[(i << 1)];
470  const LeafType high = val_corners_[(i << 1) + 1];
471  const LeafType val = axis_pos*high + (1.0f-axis_pos)*low;
472  val_corners_[i] = val;
473  }
474  val_corners_.resize(num_out);
