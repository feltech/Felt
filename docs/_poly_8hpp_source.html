198  void spx(const VecDi& pos)
199  {
200  // TODO: this is here for consistency only, since the marching
201  // cubes implementation marches in the negative z-axis, but
202  // positive x and y axes. Hence an offset is required so that the
203  // negative z-axis marching is compensated by shifting the
204  // calculation in the +z direction by one grid node.
205  // (NOTE: has no effect for 2D).
206  const VecDi pos_calc = pos - GeomImpl::SpxGridPosOffset;
207 
208  // Get corner inside-outside bitmask at this position.
209  const unsigned short mask = this->mask(pos_calc);
210  // Array of indices of zero-crossing vertices along each axis from
211  // this corner.
212  ListIdx vtx_idxs[GeomImpl::num_edges];
213  // Look up the edges that are crossed from the corner mask.
214  unsigned short vtx_mask = GeomImpl::vtx_mask[mask];
215  const short* vtx_order = GeomImpl::vtx_order[mask];
216 
217  // Cube corners are all inside or all outside.
218  if (vtx_order[0] == -1)
219  return;
220 
221  // Loop over each crossed edge in the cube, looking up
222  // (or calculating, if unavailable) the vertices at the
223  // zero-crossing.
224  for (ListIdx edge_idx = 0; edge_idx < GeomImpl::num_edges; edge_idx++ )
225  {
226  // Check if current edge is crossed by the zero curve.
227  if ((vtx_mask >> edge_idx) & 1)
228  {
229  const Edge& edge = GeomImpl::edges[edge_idx];
230  // Edges are defined as an axis and an offset.
231  // Look up index of vertex along current edge.
232  vtx_idxs[edge_idx] = idx(pos_calc + edge.offset, edge.axis);
233  }
234  }
235 
236  // Check for degenerates. Compare every calculated vertex to every
237  // other to ensure they are not located on top of one-another.
238  // E.g. corners that lie at precisely zero will have D vertices that
239  // all lie on that corner.
240  // TODO: can't just throw away all triangles like this - others may
241  // be valid - must do a per-simplex degenerate check. Maybe not
242  // worth it, though.
243 // for (UINT edge_idx1 = 0; edge_idx1 < PolyBase<D>::num_edges - 1;
244 // edge_idx1++
245 // ) {
246 // for (UINT edge_idx2 = edge_idx1+1;
247 // edge_idx2 < PolyBase<D>::num_edges; edge_idx2++)
248 // {
249 // // Check both edges are bisected by the zero-curve.
250 // if (!(((vtx_mask >> edge_idx1) & 1)
251 // && ((vtx_mask >> edge_idx2) & 1)))
252 // continue;
253 //
254 // // Get the position vector component of the vertex
255 // // information for both edges.
256 // const VecDf& pos1 = this->vtx(vtx_idxs[edge_idx1]).pos;
257 // const VecDf& pos2 = this->vtx(vtx_idxs[edge_idx2]).pos;
258 // const FLOAT dist = (pos1 - pos2).squaredNorm();
259 // // If they are essentially the same vertex,
260 // // then there is no simplex for this cube.
261 // if (dist <= ThisType::epsilon())
262 // return;
263 // }
264 // }
265 
266  // Join the vertices along each edge that the surface crosses to
267  // make a simplex (or simplices).
268  // The vtx_order lookup translates corner in-out mask to CCW
269  // vertex ordering. We take D elements at a time from the lookup,
270  // with each successive subset of D elements forming the next
271  // simplex.
272  for (ListIdx order_idx = 0; vtx_order[order_idx] != -1; order_idx += t_dims )
273  {
274  Simplex simplex;
275  // A simplex for number of dimensions D has D vertices,
276  // i.e. D endpoints.
277  for (Dim endpoint = 0; endpoint < t_dims; endpoint++)
278  {
279  // Each vertex of the simplex is stored as an index
280  // reference into the 'global' vertex array.
281  simplex.idxs(endpoint) = vtx_idxs[ vtx_order[order_idx+ListIdx(endpoint)] ];
282  }
283  // Append the simplex to the list of simplices that make up the
284  // polygonisation of this grid location.
285  m_a_spx.push_back(std::move(simplex));
286  }
287  } // End spx.

296  ListIdx idx(const VecDi& pos_a, const Dim axis)
297  {
298  // Check lookup to see if vertex has already been calculated.
299  const ListIdx idx_lookup = this->get(pos_a)(axis);
300  if (idx_lookup != Felt::null_idx) {
301  return idx_lookup;
302  }
303 
304  // Position of opposite endpoint.
305  VecDi pos_b(pos_a);
306  pos_b(axis) += 1;
307 
308  // Value of isogrid at each endpoint of this edge.
309  const Distance val_a = m_pisogrid->get(pos_a);
310  const Distance val_b = m_pisogrid->get(pos_b);
311 
312  // The newly created vertex.
313  Vertex vtx;
314 
315  // Check if lies very close to an endpoint or midpoint, if so then no need (and possibly
316  // dangerous) to interpolate.
317  if (std::abs(val_a) <= epsilon) {
318  vtx = Vertex(m_pisogrid, pos_a);
319  } else if (std::abs(val_b) <= epsilon) {
320  vtx = Vertex(m_pisogrid, pos_b);
321  } else {
322  Distance mu;
323 
324  // If close to midpoint then put at midpoint.
325  if (std::abs(val_a - val_b) <= epsilon) {
326  mu = Distance(0.5f);
327  } else
328  // Otherwise interpolate between endpoints.
329  {
330  mu = val_a / (val_a - val_b);
331  }
332 
333  const VecDf vec_a = pos_a.template cast<FLOAT>();
334  const VecDf vec_b = pos_b.template cast<FLOAT>();
335  const VecDf vec_c = vec_a + (vec_b - vec_a) * mu;
336 
337  vtx = Vertex(m_pisogrid, vec_c);
338  }
339 
340  // Append newly created vertex to the cache and return a reference to it.
341  const ListIdx idx = m_a_vtx.size();
342  m_a_vtx.push_back(std::move(vtx));
343  this->get(pos_a)(axis) = idx;
344  this->lookup().track(this->lookup().index(pos_a));
345  return idx;
346  }

356  unsigned short mask (const VecDi& pos_) const
357  {
358  // Num corners == 2^D. That is, 4 for 2D, 8 for 3D.
359  unsigned short mask = 0;
360  const ListIdx num_corners = (1 << t_dims);
361  for (ListIdx idx = 0; idx < num_corners; idx++)
362  {
363  const VecDi corner = pos_ + GeomImpl::corners[idx];
364  const Distance val = m_pisogrid->get(corner);
365  mask |= (unsigned short)((val > 0) << idx);
366  }
367  return mask;
368  }
