491  const TupleIdx list_idx_
492  ) {
493  Child& child = pself->children().get(pos_idx_child_);
494 
495  // Untrack position in child sub-grid.
496  child.lookup().untrack(pos_idx_leaf_, list_idx_);
497  child.set(pos_idx_leaf_, background_);
498  const PosIdxList& pos_idxs = child.lookup().list(list_idx_);
499 
500  // If tracking list is empty in child, untrack parent. No race condition here, as long as
501  // we stick to one thread per child rule.
502  if (pos_idxs.size() == 0)
503  {
504  // Scope for lock.
505  {
506  // Mutex lock for children grid.
507  std::lock_guard<std::mutex> lock(pself->mutex_children());
508  // Untrack this list in children grid.
509  pself->children().lookup().untrack(pos_idx_child_, list_idx_);
510  }
511 
512  // If no position is being tracked at all in any tracking list, then deactivate
513  // child. Otherwise just reset to background value.
514  if (not pself->children().lookup().is_tracked(pos_idx_child_))
515  child.deactivate(background_);
516  }
517  }
518 
519  void retrack(
520  const PosIdx pos_idx_child_, const PosIdx pos_idx_leaf_, const TupleIdx list_idx_from_,
521  const TupleIdx list_idx_to_
522  ) {
523  Child& child = pself->children().get(pos_idx_child_);
524 
525  #ifdef FELT_DEBUG_ENABLED
526  if (not pself->children().lookup().is_tracked(pos_idx_child_))
527  {
528  std::stringstream strs;
529  strs << "Attempting to move lists within an inactive child: " <<
530  format(child.index(pos_idx_leaf_)) << " from list " << list_idx_from_ <<
531  " to list " << list_idx_to_ << " in partition " <<
532  format(pself->children().index(pos_idx_child_));
533  std::string str = strs.str();
534  throw std::domain_error(str);
535  }
536  #endif
537 
538  // Move position between tracking lists in child grid.
539  child.lookup().untrack(pos_idx_leaf_, list_idx_from_);
540  child.lookup().track(pos_idx_leaf_, list_idx_to_);
541 
542  // If child is not tracked by target list or child should be untracked by source list,
543  // then apply mutex and track/untrack as necessary. No race condition, provided we stick to
544  // one child per thread rule.
545  if (
546  not pself->children().lookup().is_tracked(pos_idx_child_, list_idx_to_) ||
547  child.lookup().list(list_idx_from_).size() == 0
548  ) {
549  // Mutex lock for children grid modifications.
550  std::lock_guard<std::mutex> lock(pself->mutex_children());
551  FELT_DEBUG(child.assert_pos_idx_bounds(pos_idx_leaf_, "retrack"));
552  // Ensure parent grid tracks this child in target list.
553  pself->children().lookup().track(pos_idx_child_, list_idx_to_);
554  // If child's source list is now empty, stop tracking in parent grid.
555  if (child.lookup().list(list_idx_from_).size() == 0)
556  pself->children().lookup().untrack(pos_idx_child_, list_idx_from_);
557  }
558  }
