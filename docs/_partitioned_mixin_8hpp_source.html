463  const TupleIdx list_idx_
464  ) {
465  ChildType& child = pself->children().get(pos_idx_child_);
466 
467  // Untrack position in child sub-grid.
468  child.lookup().untrack(pos_idx_leaf_, list_idx_);
469  child.set(pos_idx_leaf_, background_);
470  const PosIdxList& pos_idxs = child.lookup().list(list_idx_);
471 
472  // If tracking list is empty in child, untrack parent. No race condition here, as long as
473  // we stick to one thread per child rule.
474  if (pos_idxs.size() == 0)
475  {
476  // Scope for lock.
477  {
478  // Mutex lock for children grid.
479  std::lock_guard<std::mutex> lock(pself->mutex_children());
480  // Untrack this list in children grid.
481  pself->children().lookup().untrack(pos_idx_child_, list_idx_);
482  }
483 
484  // If no position is being tracked at all in any tracking list, then deactivate
485  // child. Otherwise just reset to background value.
486  if (not pself->children().lookup().is_tracked(pos_idx_child_))
487  child.deactivate(background_);
488  }
489  }
490 
491  void retrack(
492  const PosIdx pos_idx_child_, const PosIdx pos_idx_leaf_, const TupleIdx list_idx_from_,
493  const TupleIdx list_idx_to_
494  ) {
495  ChildType& child = pself->children().get(pos_idx_child_);
496 
497  #ifdef FELT_DEBUG_ENABLED
498  if (not pself->children().lookup().is_tracked(pos_idx_child_))
499  {
500  std::stringstream strs;
501  strs << "Attempting to move lists within an inactive child: " <<
502  format(child.index(pos_idx_leaf_)) << " from list " << list_idx_from_ <<
503  " to list " << list_idx_to_ << " in partition " <<
504  format(pself->children().index(pos_idx_child_));
505  std::string str = strs.str();
506  throw std::domain_error(str);
507  }
508  #endif
509 
510  // Move position between tracking lists in child grid.
511  child.lookup().untrack(pos_idx_leaf_, list_idx_from_);
512  child.lookup().track(pos_idx_leaf_, list_idx_to_);
513 
514  // If child is not tracked by target list or child should be untracked by source list,
515  // then apply mutex and track/untrack as necessary. No race condition, provided we stick to
516  // one child per thread rule.
517  if (
518  not pself->children().lookup().is_tracked(pos_idx_child_, list_idx_to_) ||
519  child.lookup().list(list_idx_from_).size() == 0
520  ) {
521  // Mutex lock for children grid modifications.
522  std::lock_guard<std::mutex> lock(pself->mutex_children());
523  FELT_DEBUG(child.assert_pos_idx_bounds(pos_idx_leaf_, "retrack"));
524  // Ensure parent grid tracks this child in target list.
525  pself->children().lookup().track(pos_idx_child_, list_idx_to_);
526  // If child's source list is now empty, stop tracking in parent grid.
527  if (child.lookup().list(list_idx_from_).size() == 0)
528  pself->children().lookup().untrack(pos_idx_child_, list_idx_from_);
529  }
530  }
