171  void seed (const VecDi& pos_centre)
172  {
173  // Width of seed.
174  const VecDi& vec_width = VecDi::Constant(INT(s_layer_max));
175 
176  // Min and max positions affected by placing seed point.
177  const VecDi& pos_min = pos_centre - vec_width;
178  const VecDi& pos_max = pos_centre + vec_width;
179 
180  // Get vector size of window formed by pos_min and pos_max.
181  const VecDi& pos_window_size = pos_max - pos_min + VecDi::Constant(1); //+1 for zero coord.
182 
183  // Calculate number of grid points to be cycled through within window.
184  const PosIdx pos_idx_max = PosIdx(pos_window_size.prod());
185 
186  // Cycle through each point in window.
187  for (PosIdx pos_idx = 0; pos_idx <= pos_idx_max; pos_idx++)
188  {
189  // Calculate vector position from integer index, using utility function.
190  VecDi pos = Felt::index<D>(pos_idx, pos_window_size);
191  // Translate position into isogrid grid space.
192  pos += pos_min;
193  // Calculate vector distance from this position to seed centre.
194  const VecDi& vec_dist = pos - pos_centre;
195  // Sum of absolute distance along each axis == city-block distance.
196  const Distance dist = Distance(vec_dist.template lpNorm<1>());
197  const LayerId layer_id_pos = layer_id(dist);
198  // Check distance indicates that this point is within the narrow band.
199  if (inside_band(layer_id_pos))
200  {
201  // Append point to a narrow band layer (if applicable).
202  m_grid_isogrid.track(dist, pos, layer_idx(layer_id_pos));
203  }
204  }

272  // Partition containing lower point of bounding box, bounded by grid.
273  const VecDi& pos_child_lower =
274  pos_grid_child_lower.cwiseMax(m_grid_isogrid.pos_child(pos_leaf_lower_));
275  // Partition containing upper point of bounding box, bounded by grid.
276  const VecDi& pos_child_upper =
277  pos_grid_child_upper.cwiseMin(m_grid_isogrid.pos_child(pos_leaf_upper_));
278  // Size of bounding box at partition level.
279  const VecDi& child_bounding_box_size = pos_child_upper - pos_child_lower + one;
280  // Upper bound of leaf (1 more than upper point), bounded by grid..
281  const VecDi& pos_leaf_upper_bound = pos_grid_upper.cwiseMin(pos_leaf_upper_ + one);
282  // Upper index of bounding box.

285  update_start();

292  child_idx, child_bounding_box_size
293  );
294  const VecDi& pos_child = pos_child_without_offset + pos_child_lower;
295  const PosIdx pos_idx_child = m_grid_isogrid.children().index(pos_child);
296  typename IsoGrid::ChildType& child = m_grid_isogrid.children().get(pos_idx_child);

298  // Loop all zero-layer points within this partition.
299  for (const PosIdx pos_idx_leaf : child.lookup().list(layer_idx(0)))
300  {
301  const VecDi& pos_leaf = child.index(pos_idx_leaf);
302  // Skip zero-layer points not within finer-grained bounding box.
303  if (IsoGrid::inside(pos_leaf, pos_leaf_lower_, pos_leaf_upper_bound))

306 
307  #if defined(FELT_EXCEPTIONS) || !defined(NDEBUG)

334  update_end();

407  VecDf ray(const VecDf& pos_origin_, const VecDf& dir_) const
408  {
409 
410  using ChildHits = std::vector<ChildHit>;
411 
412  // If ray is cast from within isogrid grid, first check child grid containing origin point.
413  if (m_grid_isogrid.inside(pos_origin_))
414  {
415  const VecDf& pos_hit = ray(
416  pos_origin_, dir_,
417  m_grid_isogrid.children().get(
418  m_grid_isogrid.pos_child(pos_origin_.template cast<INT>())
419  )
420  );
421  if (pos_hit != s_ray_miss)
422  return pos_hit;
423  }
424 
425  // Ray to test against.
426  Line line(pos_origin_, dir_);
427 
428  // Tracking list for child grids that are hit.
429  ChildHits child_hits;
430 
431  // Cycle each axis, casting ray to child grid planes marching away from origin.
432  for (Dim dim = 0; dim < dir_.size(); dim++)
433  {
434  // Direction +/-1 along this axis.
435  Distance dir_dim = Distance(sgn(dir_(dim)));
436  if (dir_dim == 0)
437  continue;
438 
439  // Get next child plane along this axis.
440  Distance pos_plane_dim = round_to_next(
441  dim, dir_dim, pos_origin_(dim), m_grid_isogrid.child_size()
442  );
443 
444  // Construct vector with elements not on this axis at zero.
445  VecDf pos_plane = VecDf::Constant(0);
446  pos_plane(dim) = pos_plane_dim;
447 
448  // If the zero point on this plane is not within the grid, then jump to max/min point
449  // on isogrid grid.
450  if (!m_grid_isogrid.inside(pos_plane))
451  {
452  Distance pos_grid_dim;
453  // If casting in -'ve direction, get maximum extent.
454  if (dir_dim == -1)
455  {
456  pos_grid_dim = Distance(
457  m_grid_isogrid.offset()(dim) + m_grid_isogrid.size()(dim)
458  );
459  if (pos_plane_dim < pos_grid_dim)
460  continue;
461  }
462  // Else if casting in +'ve direction, get minimum extent.
463  else
464  {
465  pos_grid_dim = Distance(m_grid_isogrid.offset()(dim));
466  if (pos_plane_dim > pos_grid_dim)
467  continue;
468  }
469  // Reset plane position to max/min extent as calculated above.
470  pos_plane(dim) = pos_grid_dim;
471  }
472 
473  // Plane normal is opposite to ray direction.
474  VecDf normal = VecDf::Constant(0);
475  normal(dim) = -dir_dim;
476 
477  // Cast ray to plane and track any child grids hit on the way to tracking list.
478  // If child size is not a factor of grid size then this first cast could be to outside
479  // the grid. So cannot quit early here and must try next child.
480  !ray_check_track_child(child_hits, line, Plane(normal, pos_plane(dim) * dir_dim));
481 
482  // Round up/down to next child, in case we started at inexact modulo of child grid size
483  // (i.e. when isogrid grid size is not integer multiple of child grid size).
484  pos_plane_dim = round_to_next(
485  dim, dir_dim, pos_plane(dim), m_grid_isogrid.child_size()
486  );
487  // If rounding produced a different plane, then cast to that plane, and potentially track
488  // child grid to tracking list.
489  if (pos_plane_dim != pos_plane(dim))
490  {
491  pos_plane(dim) = pos_plane_dim;
492  if (!ray_check_track_child(child_hits, line, Plane(normal, pos_plane(dim) * dir_dim)))
493  continue;
494  }
495 
496  // Keep marching along planes, casting ray to each and tracking any candidate child
497  // grids to the tracking list.
498  const Distance child_size_dim = Distance(m_grid_isogrid.child_size()(dim));
499  while (true)
500  {
501  pos_plane(dim) += dir_dim * child_size_dim;
502  if (
503  !ray_check_track_child(
504  child_hits, line, Plane(normal, pos_plane(dim) * dir_dim)
505  )
506  ) {
507  break;
508  }
509  }
510  }
511 
512  // untrack children whose raycast hit point lies outside themselves. Possible if a child
513  // has been allowed to be tracked in order for the loop above to continue on to the next
514  // child, via dir vs. offset/size check in ray_check_track_child().
515 // child_hits.erase(std::untrack_if(
516 // child_hits.begin(), child_hits.end(),
517 // [this](const ChildHit& child_hit) {
518 // return !this->m_grid_isogrid.children()
519 // .get(child_hit.pos_child)
520 // .inside(child_hit.pos_intersect);
521 // }
522 // ), child_hits.end());
523 
524  // Sort candidate child grids in distance order from front to back.
525  std::sort(
526  child_hits.begin(), child_hits.end(),
527  [&pos_origin_](const ChildHit& a, const ChildHit& b) -> bool {
528  const VecDf& dist_a = (a.pos_intersect - pos_origin_);
529  const VecDf& dist_b = (b.pos_intersect - pos_origin_);
530  return dist_a.squaredNorm() < dist_b.squaredNorm();
531  }
532  );
533  // untrack any duplicate child grids from the sorted list (i.e. where ray intersects
534  // precisely at the interesction of two or more planes).
535  child_hits.erase(std::unique(
536  child_hits.begin(), child_hits.end(),
537  [](const ChildHit& a, const ChildHit& b) -> bool {
538  return a.pos_idx_child == b.pos_idx_child;
539  }
540  ), child_hits.end());
541 
542  // For each candidate child, cast ray through until the zero-curve is hit.
543  for (const ChildHit& child_hit : child_hits)
544  {
545  const VecDf& pos_hit = ray(
546  child_hit.pos_intersect, dir_,
547  m_grid_isogrid.children().get(child_hit.pos_idx_child)
548  );
549 
550  if (pos_hit != s_ray_miss)
551  return pos_hit;
552  }
553 
554  return s_ray_miss;
555  }

649  {

651 
652  // Loop over delta isogrid modified zero-layer points tracking to
653  // tracking grid.
654 
655  // Loop spatial partitions of delta for zero-layer.
656  for (
657  const PosIdx pos_idx_child
658  : m_grid_delta.children().lookup().list(layer_idx_zero))
659  {
660  // Loop leaf grid nodes with spatial partition
661  for (
662  const PosIdx pos_idx_leaf
663  : m_grid_delta.children().get(pos_idx_child).list(layer_idx_zero)
664  )
665  // track zero-layer point to tracking grid.
666  m_grid_affected.track(pos_idx_child, pos_idx_leaf, layer_idx_zero);
667  }

670  // tracking grid.
671  std::array<std::vector<ListIdx>, s_num_layers> aidx_first_neigh;
672  std::array<std::vector<ListIdx>, s_num_layers> aidx_last_neigh;
673 
674  // Loop round L times, searching outward for affected outer layer grid nodes.
675  for (LayerId udist = 1; udist <= s_layer_max; udist++)
676  {
677  // Reset the first and last element indices for each spatial partition in each layer.
678 
679  for (LayerId layer_id = s_layer_min; layer_id <= s_layer_max; layer_id++)
680  {
681  const TupleIdx layer_idx = this->layer_idx(layer_id);
682  // Get number of spatial partitions for this layer.
683  const ListIdx num_childs =
684  m_grid_affected.children().lookup().list(layer_idx).size();
685  // Resize spatial partition index lists for this layer to
686  // to include any newly tracked partitions.
687  aidx_last_neigh[ListIdx(layer_idx)].resize(num_childs);
688  // Will initialise to zero, so no further work needed for
689  // these new indices giving the start of the range.
690  aidx_first_neigh[ListIdx(layer_idx)].resize(num_childs);
691  // The final index needs to be copied from the current size
692  // of each spatial partition, so loop over partitions,
693  // copying their size into the respective last index list.
694  for (
695  ListIdx list_idx_child = 0; list_idx_child < num_childs; list_idx_child++
696  ) {
697  // Get position of this spatial partition in parent
698  // lookup grid.
699  const PosIdx pos_idx_child =
700  m_grid_affected.children().lookup().list(layer_idx)[list_idx_child];
701  // Copy number of active grid nodes for this partition
702  // into relevant index in the list.
703  aidx_last_neigh[ListIdx(layer_idx)][list_idx_child] =
704  m_grid_affected.children().get(pos_idx_child).list(layer_idx).size();
705  }
706  }
707 
708  // Loop each layer finding the affected outer layer points
709  // for each partition using the start and end points cached
710  // above.
711 
712  for (LayerId layer_id = s_layer_min; layer_id <= s_layer_max; layer_id++)
713  {
714  const TupleIdx layer_idx = this->layer_idx(layer_id);
715 
716  // Loop over spatial partitions, ignoring newly tracked ones
717  // since we're using the cached spatial partition list as
718  // the end of the range.
719  for (
720  ListIdx idx_child = 0;
721  idx_child < aidx_first_neigh[ListIdx(layer_idx)].size();
722  idx_child++
723  ) {
724  // Get position of this spatial partition in this layer.
725  const PosIdx pos_idx_child = (
726  m_grid_affected.children().lookup().list(layer_idx)[idx_child]
727  );
728 
729  // Loop over leaf grid nodes within this spatial partition, using the cached
730  // start and end indices, so that newly tracked points are skipped.
731  for (
732  ListIdx idx_neigh = aidx_first_neigh[ListIdx(layer_idx)][idx_child];
733  idx_neigh < aidx_last_neigh[ListIdx(layer_idx)][idx_child];
734  idx_neigh++
735  ) {
736  using AffectedChild = typename AffectedLookupGrid::ChildType;
737  const AffectedChild& child = m_grid_affected.children().get(pos_idx_child);
738  // This leaf grid node is the centre to search about.
739  const PosIdx pos_idx_centre = child.list(layer_idx)[idx_neigh];
740  const VecDi pos_centre = child.index(pos_idx_centre);
741 
742  // Cycle neighbours and record them if they are within the narrow band.
743  m_grid_isogrid.neighs(
744  pos_centre,
745  [this](const VecDi& pos_neigh) {
746  // Calculate layer of this neighbouring point from the isogrid grid.
747  const LayerId layer_id_neigh = this->layer_id(pos_neigh);
748  // If the calculated layer lies within the narrow band, then we
749  // want to track it.
750  if (inside_band(layer_id_neigh))
751  {
752  // track the neighbour point to the
753  // tracking grid. Will reject
754  // duplicates.
755  m_grid_affected.track(
756  pos_neigh, this->layer_idx(layer_id_neigh)
757  );
758  }
759  }
760  ); // End neighbourhood query.
761  } // End for leaf grid node.
762  } // End for spatial partition.
763  } // End for layer.
764 
765  // Now we've found the neighbours of the pre-existing grid
766  // points in the tracking list, we want to skip these on the
767  // next loop, so set the start index for each partition of
768  // each layer to the previous end index.
769 
770  for (LayerId layer_id = s_layer_min; layer_id <= s_layer_max; layer_id++)
771  {
772  const ListIdx layer_idx = ListIdx(this->layer_idx(layer_id));
773  // Loop over spatial partitions.

788  void update_zero_layer (AffectedLookupGrid* plookup_buffer_)
789  {
790  using DeltaChild = typename DeltaIsoGrid::ChildType;
791  using IsoChild = typename IsoGrid::ChildType;
792 
793  const TupleIdx layer_idx_zero = layer_idx(0);

795

800  {

808  const Distance iso_delta = delta_child.get(pos_idx_leaf);
809  const Distance iso_new = iso_prev + iso_delta;
810  const LayerId layer_id_new = layer_id(iso_new);
811 
812  #ifdef FELT_DEBUG_ENABLED
813

821  " + " << iso_delta << " = " << iso_new;
822  std::string str = strs.str();
823  throw std::domain_error(str);
824  }
825 
826  if (
827  std::abs(layer_id_new) != 0
828  && std::abs(layer_id_new) != 1
829  && std::abs(layer_id_new) != -1

832  strs << "Zero layer update out of bounds. Attempting to change value at" <<
833  Felt::format(isogrid_child.index(pos_idx_leaf)) << " to " << iso_new <<
834  " would give a layer of " << layer_id_new <<
835  ", which is too much of a jump";
836  std::string str = strs.str();
837  throw std::domain_error(str);
838  }
839 
840  #endif
841 
842  // Update value in grid with new signed distance.
843  isogrid_child.set(pos_idx_leaf, iso_new);
844  // Potentially track to status change, if narrow band layer has changed.
845  status_change(pos_idx_child, pos_idx_leaf, 0, layer_id_new, plookup_buffer_);

898  template <typename GridType>
899  bool update_distance(
900  const LayerId layer_id_, const LayerId side_,
901  const GridType* plookup_, AffectedLookupGrid* plookup_buffer_
902  ) {
903  using IsoChild = typename IsoGrid::ChildType;
904  using DeltaIsoChild = typename DeltaIsoGrid::ChildType;
905 
906  bool is_status_changed = false;
907 
908  const TupleIdx layer_idx = this->layer_idx(layer_id_);
909  const PosIdxList& pos_idxs_children = plookup_->children().lookup().list(layer_idx);
910  const ListIdx num_childs = pos_idxs_children.size();
911 
912  // First pass: calculate distance and add to delta isogrid.
913  FELT_PARALLEL_FOR(num_childs, firstprivate(plookup_, plookup_buffer_))
914  for (ListIdx list_idx = 0; list_idx < num_childs; list_idx++)
915  {
916  // Child spatial partition position
917  const ListIdx pos_idx_child = pos_idxs_children[list_idx];
918 
919  const PosIdxList& apos_leafs =
920  plookup_->children().get(pos_idx_child).list(layer_idx);
921 
922  // Calculate distance of every point in this layer to the zero
923  // layer, and store in delta isogrid grid.
924  // Delta isogrid grid is used to allow for asynchronous updates, that
925  // is, to prevent neighbouring points affecting the distance
926  // transform.
927  for (const PosIdx pos_idx_leaf : apos_leafs)
928  {
929  // Distance from this position to zero layer.
930  const Distance dist = distance(pos_idx_child, pos_idx_leaf, side_);
931 
932  #ifdef FELT_DEBUG_ENABLED
933 
934  const LayerId layer_id_new = layer_id(dist);
935 
936  if (
937  layer_id_new != layer_id_ &&
938  layer_id_new != layer_id_ + 1 &&
939  layer_id_new != layer_id_ - 1
940  ) {
941  const VecDi& pos =
942  m_grid_isogrid.children().get(pos_idx_child).index(pos_idx_leaf);
943  std::stringstream strs;
944  strs << "Outer layer distance update value out of bounds." << std::endl <<
945  str_neighs(pos) << " distance of " << dist <<
946  ", which is too much of a jump";
947 
948  std::string str = strs.str();
949  throw std::domain_error(str);
950  }
951 
952  #endif
953 
954  // Update delta isogrid grid.
955  m_grid_delta.children().get(pos_idx_child).track(dist, pos_idx_leaf, layer_idx);
956  }
957  }
958 
959  // Second pass: apply distance to isogrid and update status change lists.
960  FELT_PARALLEL_FOR(num_childs, firstprivate(plookup_, plookup_buffer_))
961  for (ListIdx list_idx = 0; list_idx < num_childs; list_idx++)
962  {
963  const PosIdx pos_idx_child = pos_idxs_children[list_idx];
964 
965  IsoChild& grid_isogrid_child = m_grid_isogrid.children().get(pos_idx_child);
966  DeltaIsoChild& grid_delta_child = m_grid_delta.children().get(pos_idx_child);
967 
968  const PosIdxList& pos_idxs_leafs =
969  plookup_->children().get(pos_idx_child).list(layer_idx);
970 
971  // Update distance in isogrid from delta isogrid and append any points that
972  // move out of their layer to a status change list.
973  for (const PosIdx pos_idx_leaf : pos_idxs_leafs)
974  {
975  // Distance calculated above.
976  const Distance dist = grid_delta_child.get(pos_idx_leaf);
977  const LayerId layer_id_new = layer_id(dist);
978 
979  #ifdef FELT_DEBUG_ENABLED
980 
981  if (
982  layer_id_new != layer_id_ &&
983  layer_id_new != layer_id_ + 1 &&
984  layer_id_new != layer_id_ - 1
985  ) {
986  const VecDi& pos =
987  m_grid_isogrid.children().get(pos_idx_child).index(pos_idx_leaf);
988  std::stringstream strs;
989  strs << "Outer layer distance update value out of bounds. Attempting to" <<
990  " move " << Felt::format(pos) << " in layer " << layer_id_ << " to a" <<
991  " distance of " << dist << " would result in a layer of " <<
992  layer_id_new << ", which is too much of a jump" ;
993  std::string str = strs.str();
994  throw std::domain_error(str);
995  }
996 
997  #endif
998 
999  grid_isogrid_child.set(pos_idx_leaf, dist);
1000  is_status_changed |= status_change(
1001  pos_idx_child, pos_idx_leaf, layer_id_, layer_id_new, plookup_buffer_
1002  );
1003  }
1004 
1005  }
1006 
1007  return is_status_changed;
1008 
1009  } // End update_distance.

1022  bool status_change (
1023  const PosIdx pos_idx_child_, const PosIdx pos_idx_leaf_, const LayerId layer_id_from_,
1024  const LayerId layer_id_to_, AffectedLookupGrid* plookup_buffer_
1025  ) {
1026  using StatusChangeChild = typename StatusChangeGrid::ChildType;
1027 
1028  if (layer_id_from_ == layer_id_to_)
1029  return false;
1030 
1031  #ifdef FELT_DEBUG_ENABLED
1032  m_grid_isogrid.children().assert_pos_idx_bounds(pos_idx_child_, "status_change child: ");
1033  m_grid_isogrid.children().get(pos_idx_child_).assert_pos_idx_bounds(
1034  pos_idx_leaf_, "status_change leaf: "
1035  );
1036  #endif
1037 
1038  StatusChangeChild& child = m_grid_status_change.children().get(pos_idx_child_);
1039  LayerId layer_id_to = child.get(pos_idx_leaf_);
1040 
1041  // If the position is already marked for status change, this must be a subsequent loop
1042  // around `converge_distance`, and this leaf position is "jumping" more than one layer.
1043  if (layer_id_to != s_outside)
1044  {
1045  child.set(pos_idx_leaf_, layer_id_to_);
1046  }
1047  else
1048  {
1049  m_grid_status_change.track(
1050  layer_id_to_, pos_idx_child_, pos_idx_leaf_, layer_idx(layer_id_from_)
1051  );
1052  }
1053 
1054  // Keep a record of points that undergo a status change, since these will have to have
1055  // their distance calculated again, and again, until they no longer status change. This
1056  // is required for collapsing regions where the surface should disappear because the
1057  // zero-layer is gone.
1058  if (inside_band(layer_id_to_))
1059  plookup_buffer_->track(pos_idx_child_, pos_idx_leaf_, layer_idx(layer_id_to_));
1060 
1061  return true;
1062  }

1273  Distance distance (VecDi pos_, Distance dist_, const Distance dir_) const
1274  {
1275  #ifdef FELT_DEBUG_ENABLED
1276  const VecDi pos_original(pos_);
1277  #endif
1278  // Transform to unsigned distance.
1279  dist_ *= dir_;
1280 
1281  // Get neighbouring point that is next closest to the zero-layer.
1282  m_grid_isogrid.neighs(
1283  pos_,
1284  [this, &pos_, &dist_, dir_](const VecDi& pos_neigh_) {
1285  if (not m_grid_isogrid.inside(pos_neigh_))
1286  return;
1287  const Distance dist_neigh = m_grid_isogrid.get(pos_neigh_);
1288  // Check absolute value of this neighbour is less than nearest point.
1289  // Multiplying by `dir` has two effects:
1290  // - It has the same effect as abs() for points on the same side of the band.
1291  // - It ensures points on the opposite side are negative, so that < comparison
1292  // prefers those points, which is good because we're interested in the neighbour
1293  // in the *direction* of the zero-curve.
1294  if (dist_neigh*dir_ < dist_) {
1295  pos_ = pos_neigh_;
1296  dist_ = dist_neigh*dir_;
1297  }
1298  }
1299  );
1300 
1301  // This point's distance is then the distance of the closest neighbour +/-1, depending
1302  // which side of the band we are looking at. So first transform back into signed distance
1303  // then add +/-1.
1304  const Distance dist_neigh = dist_*dir_;
1305  dist_ = dist_neigh + dir_;
1306 
1307  #ifdef FELT_DEBUG_ENABLED
1308  const LayerId layer_id_pos = layer_id(pos_original);
1309  const LayerId layer_id_neigh = layer_id(dist_neigh);
1310 
1311  if (
1312  std::abs(layer_id_pos) < std::abs(layer_id_neigh) &&
1313  sgn(layer_id_pos) == sgn(layer_id_neigh)
1314  ) {
1315  std::stringstream sstr;
1316  sstr << "Neighbour closest to zero curve is further away than source position: " <<
1317  Felt::format(pos_original) << " at " << m_grid_isogrid.get(pos_original) <<
1318  " is closer than " << Felt::format(pos_) << " at " << m_grid_isogrid.get(pos_) <<
1319  " but should not be";
1320  std::string str = sstr.str();
1321  throw std::domain_error(str);
1322  }
1323 
1324  #endif
1325 
1326  return dist_;
1327  }

1338  VecDf pos_sample, const VecDf& dir, const typename IsoGrid::ChildType& child

1341

1388  break;
1389 
1390  normal = m_grid_isogrid.grad(pos_sample);
1391  normal.normalize();
1392  }
1393

1401 // << " sample(" << Felt::format(pos_sample) << ")"
1402 // << " at dist " << dist;
1403  }
1404  #endif
1405 
1406  return pos_sample;
1407  }
1408  }
1409 
1410  t_leaf += 0.5f;
1411 
1412  pos_sample = line_leaf.pointAt(t_leaf);
1413  } // End while inside child grid.
1414 
1415  return s_ray_miss;
